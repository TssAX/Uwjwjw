Tabs.Home:CreateToggle("WalkOn", {
    Title = "Walk On Water",
    Description = "Create invisible platform to walk on water",
    Default = false,
    Callback = function(Value)
        local parts = {}
        local partSize = 2048
        local totalDistance = 50000
        local startPosition = Vector3.new(-2, -9.5, -2)
        local numberOfParts = math.ceil(totalDistance / partSize)

        local function createParts()
            for x = 0, numberOfParts - 1 do
                for z = 0, numberOfParts - 1 do
                    local positions = {
                        {pos = Vector3.new(x * partSize, 0, z * partSize), name = "Side"},
                        {pos = Vector3.new(-x * partSize, 0, z * partSize), name = "LeftRight"},
                        {pos = Vector3.new(-x * partSize, 0, -z * partSize), name = "UpLeft"},
                        {pos = Vector3.new(x * partSize, 0, -z * partSize), name = "UpRight"}
                    }
                    
                    for _, posData in ipairs(positions) do
                        local part = Instance.new("Part")
                        part.Size = Vector3.new(partSize, 1, partSize)
                        part.Position = startPosition + posData.pos
                        part.Anchored = true
                        part.Transparency = 1
                        part.CanCollide = true
                        part.Name = "WaterPart_" .. posData.name .. "_" .. x .. "_" .. z
                        part.Parent = workspace
                        table.insert(parts, part)
                    end
                end
            end
        end

        local function cleanupParts()
            for _, part in ipairs(parts) do
                if part and part.Parent then
                    part:Destroy()
                end
            end
            parts = {}
        end

        if Value then
            createParts()
        else
            cleanupParts()
        end
    end
})

Tabs.Home:CreateToggle("AutoClearInventory", {
    Title = "Auto Clear Inventory",
    Description = "Automatically consume all boosts in inventory",
    Default = false,
    Callback = function(Value)
        local autoEatBoostsEnabled = Value

        local boostsList = {
            "ULTRA Shake",
            "TOUGH Bar",
            "Protein Shake",
            "Energy Shake",
            "Protein Bar",
            "Energy Bar",
            "Tropical Shake"
        }

        local function eatAllBoosts()
            local player = game.Players.LocalPlayer
            local backpack = player:WaitForChild("Backpack")
            local character = player.Character or player.CharacterAdded:Wait()

            for _, boostName in ipairs(boostsList) do
                local boost = backpack:FindFirstChild(boostName)
                while boost do
                    boost.Parent = character
                    pcall(function()
                        boost:Activate()
                    end)
                    task.wait(0)
                    boost = backpack:FindFirstChild(boostName)
                end
            end
        end

        task.spawn(function()
            while true do
                if autoEatBoostsEnabled then
                    eatAllBoosts()
                    task.wait(2)
                else
                    task.wait(1)
                end
            end
        end)
    end
})

    Title = "No Punch Animation",
    Description = "Remove punch animations",
    Default = false,
    Callback = function(Value)
        local LocalPlayer = game.Players.LocalPlayer
        
        if Value then
            local blockedAnimations = {
                ["rbxassetid://3638729053"] = true,
                ["rbxassetid://3638767427"] = true,
            }

            local function stopBlockedAnimations(humanoid)
                for _, track in pairs(humanoid:GetPlayingAnimationTracks()) do
                    if track.Animation then
                        local animId = track.Animation.AnimationId
                        local animName = track.Name:lower()
                        if blockedAnimations[animId] or animName:match("punch") or animName:match("attack") or animName:match("right") then
                            track:Stop()
                        end
                    end
                end
            end

            local function setupAnimationBlocking()
                local char = LocalPlayer.Character
                if not char or not char:FindFirstChild("Humanoid") then return end
                local humanoid = char:FindFirstChild("Humanoid")

                stopBlockedAnimations(humanoid)

                if not _G.AnimBlockConnection then
                    _G.AnimBlockConnection = humanoid.AnimationPlayed:Connect(function(track)
                        if track.Animation then
                            local animId = track.Animation.AnimationId
                            local animName = track.Name:lower()
                            if blockedAnimations[animId] or animName:match("punch") or animName:match("attack") or animName:match("right") then
                                track:Stop()
                            end
                        end
                    end)
                end
            end

            setupAnimationBlocking()

            _G.AutoPunchLoop = true
            task.spawn(function()
                while _G.AutoPunchLoop do
                    local char = LocalPlayer.Character
                    if char and char:FindFirstChild("Humanoid") then
                        stopBlockedAnimations(char.Humanoid)
                    end

                    local punch = LocalPlayer.Backpack:FindFirstChild("Punch") or (char and char:FindFirstChild("Punch"))
                    if punch then
                        if punch.Parent ~= char then
                            punch.Parent = char
                        end
                        LocalPlayer.muscleEvent:FireServer("punch", "rightHand")
                        LocalPlayer.muscleEvent:FireServer("punch", "leftHand")
                    end
                    task.wait(0.01)
                end
            end)

            if not _G.CharacterAddedConnection then
                _G.CharacterAddedConnection = LocalPlayer.CharacterAdded:Connect(function()
                    task.wait(1)
                    setupAnimationBlocking()
                end)
            end
        else
            _G.AutoPunchLoop = false
            if _G.AnimBlockConnection then 
                _G.AnimBlockConnection:Disconnect() 
                _G.AnimBlockConnection = nil 
            end
        end
    end
})

end

local function isWhitelisted(player)
    for _, name in ipairs(_G.whitelistedPlayers) do
        if name:lower() == player.Name:lower() then
            return true
        end
    end
    return false
end

local function isBlacklisted(player)
    for _, name in ipairs(_G.blacklistedPlayers) do
        if name:lower() == player.Name:lower() then
            return true
        end
    end
    return false
end

local function refreshPlayerDropdowns()
    local playerValues = {}
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            table.insert(playerValues, getPlayerDisplayText(player))
        end
    end

    if whitelistDropdown then
        whitelistDropdown:SetValues(playerValues)
    end
    if blacklistDropdown then
        blacklistDropdown:SetValues(playerValues)
    end
end

local function formatNumber(number)
	local formatted = tostring(math.floor(number))
	local k
	while true do
		formatted, k = string.gsub(formatted, "^(-?%d+)(%d%d%d)", '%1,%2')
		if k == 0 then
			break
		end
	end
	return formatted
end

local function formatTime(seconds)
	local days = math.floor(seconds / 86400)
	local hours = math.floor((seconds % 86400) / 3600)
	local minutes = math.floor((seconds % 3600) / 60)
	local secs = seconds % 60
	return string.format("%dd %dh %dm %ds", days, hours, minutes, secs)
end

local function checkEggTimer()
	local boostFolder = game.Players.LocalPlayer:FindFirstChild("boostTimersFolder")
	if not boostFolder then
		return false
	end
	local eggTimer = boostFolder:FindFirstChild("Protein Egg")
	if not eggTimer then
		return useOneEgg()
	end
	if tonumber(eggTimer.Value) <= 25 then
		return useOneEgg()
	end
	return true
end

local function createParticles(part)

		ColorSequenceKeypoint.new(1, Color3.fromRGB(0, 155, 0))
	})

		NumberSequenceKeypoint.new(0, 0.5),

	return {
